例如：
　　　PersonnelRecord ::= SET
　　　　　　{
　　　　　　name　　　Name,
　　　　　　title　　IA5String
　　　　　　number　　　INTEGER,
　　　　　　dateOfHire　　　Date, (雇用日期)
　　　　　　nameOfSpouse　　Name, (配偶姓名)
　　　　　　children　　SEQUENCE OF ChildInformation
　　　　　　}
　　　ChildInformation ::=SET
　　　　　　{
　　　　　name　　　Name,
　　　　　dateOfBirth Date
　　　　　}
　　　Name ::=SEQUENCE
　　　　　{
　　　　　givenName　　IA5String,
　　　　　initial　　IA5String,
　　　　　familyName　　IA5String
　　　　　}
　　　Date ::=IA5String
