问答第1题
　　a. select count(distinct o.ss)
　　　 from accident a, log l, owns o
　　　 where a.date = l.date and a.driver = l.driver and l.license = o.license and
　　　 format(a.date, "YYYY") = '1989'
　　 b. select count(distinct l.date, l.driver)
　　　 from person p, owns o, log l
　　　 where p.ss = o.ss and o.license = l.license and p.name = 'John Smith'
　　 c. insert into person values ('110108', 'Feng JH', 'Tsinghua University')
　　 d. delete from car
　　　　 where car.license in (select c.license
　　　　 from person as p, owns as o, car as c
　　　　 where p.ss = o.ss and o.license = c.license and
　　　　　　　 p.name = 'John Smith' and c.model = 'Mazda')
　　 e. insert into log
　　　　 select c.license, '1990.10.21', 'Dog'
　　　　 from person as p, owns as o, car as c
　　　　 where p.ss = o.ss and o.license = c.license and
　　　　　　　 p.name = 'Jones' and c.model = 'Toyota'
　　　 　insert into accident values ('1990.10.21', 'Dog', 1000)
